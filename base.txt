local hindilib = {
	create = function(instance, parent)
		local new = Instance.new(instance)
		new.Parent = parent
		return new
	end,
	readudim2 = function(table1, table2)
		return UDim2.new(table1[1], table1[2], table2[1], table2[2])
	end,
	drag = function(frame)
		local dragging, start, startpos = false, nil, nil
		frame.InputBegan:Connect(function(input)
			if input.UserInputType == Enum.UserInputType.MouseButton1 then
				dragging = true
				start = input.Position
				startpos = frame.Position
			end
		end)
		frame.InputChanged:Connect(function(input)
			if dragging and input.UserInputType == Enum.UserInputType.MouseMovement then
				local delta = input.Position - start
				frame.Position = UDim2.new(startpos.X.Scale, startpos.X.Offset + delta.X, startpos.Y.Scale, startpos.Y.Offset + delta.Y)
			end
		end)
		frame.InputEnded:Connect(function(input)
			if input.UserInputType == Enum.UserInputType.MouseButton1 then
				dragging = false
			end
		end)
	end,
	getplrs = function(box)
		local tab = {
			
		}
		if box.Text == "me" then
			return {game:GetService("Players").LocalPlayer}
		elseif box.Text == "all" then
			return game:GetService("Players"):GetPlayers()
		elseif box.Text ~= "me" and box.Text ~= "all" then
			for datamodel, plr in pairs(game:GetService("Players"):GetPlayers()) do
				if string.find(plr.Name:lower(), box.Text) == true then
					table.insert(tab, plr)
				end
			end
		end
		return tab
	end,
}
